Append/concat a matrix horizontally

Example
var M = require('./src/matrix_lib');
var A = M.range(3);
console.log(A.catH(A).value) // Show value
console.log(A.catH(A).size()) // Show size
// Returns 
//[ [ 0, 0 ], [ 1, 1 ], [ 2, 2 ] ]
//[ 3, 2 ]
